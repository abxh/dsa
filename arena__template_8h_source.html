245FUNCTION_LINKAGE void JOIN(ARENA_NAME, init)(ARENA_TYPE *self, const size_t len, unsigned char *backing_buf)
246{
247 assert(self);
248 assert(backing_buf);
249
250 const uintptr_t padding = calc_alignment_padding(alignof(max_align_t), (uintptr_t)backing_buf);
251
252 assert(len >= padding);
253
254 self->buf_ptr = &backing_buf[padding];
255 self->buf_len = len - padding;
256 self->curr_offset = 0;
257 self->prev_offset = 0;
258}

268FUNCTION_LINKAGE void *JOIN(ARENA_NAME, allocate_aligned)(ARENA_TYPE *self, const size_t alignment, const size_t size)
269{
270 assert(self);
271
272 void *ptr = (void *)&self->buf_ptr[self->curr_offset];
273
274 size_t space_left = self->buf_len - (size_t)self->curr_offset;
275
276 const bool has_space_left = align(alignment, size, &ptr, &space_left);
277 if (!has_space_left) {
278 return NULL;
279 }
280
281 const uintptr_t relative_offset = (uintptr_t)((unsigned char *)ptr - &self->buf_ptr[0]);
282
283 self->prev_offset = relative_offset;
284 self->curr_offset = relative_offset + size;
285
286 memset(ptr, 0, size);
287
288 return ptr;
289}

299static inline void *JOIN(JOIN(internal, ARENA_NAME),
300 try_optimizing_w_prev_offset)(ARENA_TYPE *self, unsigned char *old_ptr, const size_t old_size,
301 const size_t new_size)
302{
303 if (&self->buf_ptr[self->prev_offset] != old_ptr) {
304 return NULL;
305 }
306
307 self->curr_offset = self->prev_offset + new_size;
308
309 if (new_size > old_size) {
310 const size_t diff = new_size - old_size;
311
312 memset(&self->buf_ptr[self->curr_offset], 0, diff);
313 }
314
315 return old_ptr;
316}

319FUNCTION_LINKAGE void *JOIN(ARENA_NAME, reallocate_aligned)(ARENA_TYPE *self, void *old_ptr_, const size_t alignment,
320 const size_t old_size, const size_t new_size)
321{
322 assert(self);
323 assert(IS_POW2(alignment));
324
325 unsigned char *old_ptr = (unsigned char *)old_ptr_;
326
327 const bool misc_input = old_ptr == NULL || old_size == 0 || new_size == 0;
328 const bool inside_arena_buf = &self->buf_ptr[0] <= old_ptr && old_ptr <= &self->buf_ptr[self->buf_len - 1];
329 if (misc_input || !inside_arena_buf) {
330 return NULL;
331 }
332
333 const bool has_optimized_w_prev_buf =
334 JOIN(JOIN(internal, ARENA_NAME), try_optimizing_w_prev_offset)(self, old_ptr, old_size, new_size);
335 if (has_optimized_w_prev_buf) {
336 return old_ptr;
337 }
338
339 const size_t copy_size = old_size < new_size ? old_size : new_size;
340
341 void *new_mem = JOIN(ARENA_NAME, allocate_aligned)(self, alignment, new_size);
342
343 memmove(new_mem, old_ptr, copy_size);
344
345 return new_mem;
346}
