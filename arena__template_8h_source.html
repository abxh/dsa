249FUNCTION_LINKAGE void JOIN(ARENA_NAME, init)(ARENA_TYPE *self, const size_t len, unsigned char *backing_buf)
250{
251 assert(self);
252 assert(backing_buf);
253
254 const uintptr_t padding = calc_alignment_padding(alignof(max_align_t), (uintptr_t)backing_buf);
255
256 assert(len >= padding);
257
258 self->buf_ptr = &backing_buf[padding];
259 self->buf_len = len - padding;
260 self->curr_offset = 0;
261 self->prev_offset = 0;
262}

272FUNCTION_LINKAGE void *JOIN(ARENA_NAME, allocate_aligned)(ARENA_TYPE *self, const size_t alignment, const size_t size)
273{
274 assert(self);
275
276 void *ptr = (void *)&self->buf_ptr[self->curr_offset];
277
278 size_t space_left = self->buf_len - (size_t)self->curr_offset;
279
280 const bool has_space_left = align(alignment, size, &ptr, &space_left);
281 if (!has_space_left) {
282 return NULL;
283 }
284
285 const uintptr_t relative_offset = (uintptr_t)((unsigned char *)ptr - &self->buf_ptr[0]);
286
287 self->prev_offset = relative_offset;
288 self->curr_offset = relative_offset + size;
289
290 memset(ptr, 0, size);
291
292 return ptr;
293}

303static inline void *JOIN(JOIN(internal, ARENA_NAME),
304 try_optimizing_w_prev_offset)(ARENA_TYPE *self, unsigned char *old_ptr, const size_t old_size,
305 const size_t new_size)
306{
307 if (&self->buf_ptr[self->prev_offset] != old_ptr) {
308 return NULL;
309 }
310
311 self->curr_offset = self->prev_offset + new_size;
312
313 if (new_size > old_size) {
314 const size_t diff = new_size - old_size;
315
316 memset(&self->buf_ptr[self->curr_offset], 0, diff);
317 }
318
319 return old_ptr;
320}

323FUNCTION_LINKAGE void *JOIN(ARENA_NAME, reallocate_aligned)(ARENA_TYPE *self, void *old_ptr_, const size_t alignment,
324 const size_t old_size, const size_t new_size)
325{
326 assert(self);
327 assert(IS_POW2(alignment));
328
329 unsigned char *old_ptr = (unsigned char *)old_ptr_;
330
331 const bool misc_input = old_ptr == NULL || old_size == 0 || new_size == 0;
332 const bool inside_arena_buf = &self->buf_ptr[0] <= old_ptr && old_ptr <= &self->buf_ptr[self->buf_len - 1];
333 if (misc_input || !inside_arena_buf) {
334 return NULL;
335 }
336
337 const bool has_optimized_w_prev_buf =
338 JOIN(JOIN(internal, ARENA_NAME), try_optimizing_w_prev_offset)(self, old_ptr, old_size, new_size);
339 if (has_optimized_w_prev_buf) {
340 return old_ptr;
341 }
342
343 const size_t copy_size = old_size < new_size ? old_size : new_size;
344
345 void *new_mem = JOIN(ARENA_NAME, allocate_aligned)(self, alignment, new_size);
346
347 memmove(new_mem, old_ptr, copy_size);
348
349 return new_mem;
350}
