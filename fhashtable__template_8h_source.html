441FUNCTION_LINKAGE FHASHTABLE_TYPE *JOIN(FHASHTABLE_NAME, init)(FHASHTABLE_TYPE *self, const uint32_t pow2_capacity)
442{
443 assert(self);
444 assert(IS_POW2(pow2_capacity));
445
446 self->count = 0;
447 self->capacity = pow2_capacity;
448
449 for (uint32_t i = 0; i < self->capacity; i++) {
450 self->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
451 }
452
453 return self;
454}

456FUNCTION_LINKAGE FHASHTABLE_TYPE *JOIN(FHASHTABLE_NAME, create)(const uint32_t min_capacity)
457{
458 if (min_capacity == 0 || min_capacity > UINT32_MAX / 2 + 1) {
459 return NULL;
460 }
461
462 const uint32_t capacity = round_up_pow2_32(min_capacity);
463
464 if (FHASHTABLE_CALC_SIZEOF_OVERFLOWS(FHASHTABLE_NAME, capacity)) {
465 return NULL;
466 }
467
468 const uint32_t size = FHASHTABLE_CALC_SIZEOF(FHASHTABLE_NAME, capacity);
469
470 FHASHTABLE_TYPE *self = (FHASHTABLE_TYPE *)calloc(1, size);
471
472 if (!self) {
473 return NULL;
474 }
475
476 FHASHTABLE_INIT(self, capacity);
477
478 return self;
479}

502FUNCTION_LINKAGE bool JOIN(FHASHTABLE_NAME, contains_key)(const FHASHTABLE_TYPE *self, const KEY_TYPE key)
503{
504 assert(self != NULL);
505
506 const uint32_t key_hash = HASH_FUNCTION(key);
507 const uint32_t index_mask = self->capacity - 1;
508
509 uint32_t index = key_hash & index_mask;
510 uint32_t max_possible_offset = 0;
511
512 while (true) {
513 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
514
515 const bool below_max = max_possible_offset <= self->slots[index].offset;
516
517 if (!(not_empty && below_max)) {
518 break;
519 }
520
521 if (KEY_IS_EQUAL(self->slots[index].key, key)) {
522 return true;
523 }
524
525 index++;
526 index &= index_mask;
527 max_possible_offset++;
528 }
529 return false;
530}

607FUNCTION_LINKAGE void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE *self, KEY_TYPE key, VALUE_TYPE value)
608{
609 assert(self != NULL);
610 assert(FHASHTABLE_CONTAINS_KEY(self, key) == false);
611
612 const uint32_t index_mask = self->capacity - 1;
613 const uint32_t key_hash = HASH_FUNCTION(key);
614
615 uint32_t index = key_hash & index_mask;
616 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
617
618 while (true) {
619 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
620
621 if (!not_empty) {
622 break;
623 }
624
625 if (current_slot.offset > self->slots[index].offset) {
626 FHASHTABLE_SWAP_SLOTS(&self->slots[index], &current_slot);
627 }
628
629 index++;
630 index &= index_mask;
631 current_slot.offset++;
632 }
633 self->slots[index] = current_slot;
634 self->count++;
635}

637FUNCTION_LINKAGE void JOIN(FHASHTABLE_NAME, update)(FHASHTABLE_TYPE *self, KEY_TYPE key, VALUE_TYPE value)
638{
639 assert(self != NULL);
640
641 const uint32_t index_mask = self->capacity - 1;
642 const uint32_t key_hash = HASH_FUNCTION(key);
643
644 uint32_t index = key_hash & index_mask;
645 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
646
647 while (true) {
648 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
649
650 if (!not_empty) {
651 break;
652 }
653
654 const bool offset_is_same = current_slot.offset == self->slots[index].offset;
655
656 const bool key_is_equal = KEY_IS_EQUAL(current_slot.key, self->slots[index].key);
657
658 if (offset_is_same && key_is_equal) {
659 self->slots[index].value = current_slot.value;
660 return;
661 }
662
663 if (current_slot.offset > self->slots[index].offset) {
664 FHASHTABLE_SWAP_SLOTS(&current_slot, &self->slots[index]);
665 }
666
667 index++;
668 index &= index_mask;
669 current_slot.offset++;
670 }
671
672 self->slots[index] = current_slot;
673 self->count++;
674}

677static inline void JOIN(internal, JOIN(FHASHTABLE_NAME, backshift))(FHASHTABLE_TYPE *self, const uint32_t index_mask,
678 uint32_t index)
679{
680 assert(self);
681
682 uint32_t next_index = (index + 1) & index_mask;
683
684 while (true) {
685 const bool not_empty = self->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
686
687 const bool offset_is_non_zero = self->slots[next_index].offset > 0;
688
689 if (!(not_empty && offset_is_non_zero)) {
690 break;
691 }
692
693 self->slots[index] = self->slots[next_index];
694 self->slots[index].offset--;
695
696 self->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
697
698 index = next_index;
699 next_index = (index + 1) & index_mask;
700 }
701}

704FUNCTION_LINKAGE bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE *self, const KEY_TYPE key)
705{
706 assert(self != NULL);
707
708 const uint32_t index_mask = self->capacity - 1;
709 const uint32_t key_hash = HASH_FUNCTION(key);
710
711 uint32_t index = key_hash & index_mask;
712 uint32_t max_possible_offset = 0;
713
714 while (true) {
715 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
716
717 const bool below_max = max_possible_offset <= self->slots[index].offset;
718
719 if (!(not_empty && below_max)) {
720 break;
721 }
722
723 const bool key_is_equal = KEY_IS_EQUAL(key, self->slots[index].key);
724
725 if (!key_is_equal) {
726 index = (index + 1) & index_mask;
727 max_possible_offset++;
728 continue;
729 }
730
731 self->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
732 self->count--;
733
734 FHASHTABLE_BACKSHIFT(self, index_mask, index);
735
736 return true;
737 }
738 return false;
739}
