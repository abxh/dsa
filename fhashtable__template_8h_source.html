442FUNCTION_LINKAGE FHASHTABLE_TYPE *JOIN(FHASHTABLE_NAME, init)(FHASHTABLE_TYPE *self, const uint32_t pow2_capacity)
443{
444 assert(self);
445 assert(IS_POW2(pow2_capacity));
446
447 self->count = 0;
448 self->capacity = pow2_capacity;
449
450 for (uint32_t i = 0; i < self->capacity; i++) {
451 self->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
452 }
453
454 return self;
455}

457FUNCTION_LINKAGE FHASHTABLE_TYPE *JOIN(FHASHTABLE_NAME, create)(const uint32_t min_capacity)
458{
459 if (min_capacity == 0 || min_capacity > UINT32_MAX / 2 + 1) {
460 return NULL;
461 }
462
463 const uint32_t capacity = round_up_pow2_32(min_capacity);
464
465 if (FHASHTABLE_CALC_SIZEOF_OVERFLOWS(FHASHTABLE_NAME, capacity)) {
466 return NULL;
467 }
468
469 const uint32_t size = FHASHTABLE_CALC_SIZEOF(FHASHTABLE_NAME, capacity);
470
471 FHASHTABLE_TYPE *self = (FHASHTABLE_TYPE *)calloc(1, size);
472
473 if (!self) {
474 return NULL;
475 }
476
477 FHASHTABLE_INIT(self, capacity);
478
479 return self;
480}

503FUNCTION_LINKAGE bool JOIN(FHASHTABLE_NAME, contains_key)(const FHASHTABLE_TYPE *self, const KEY_TYPE key)
504{
505 assert(self != NULL);
506
507 const uint32_t key_hash = HASH_FUNCTION(key);
508 const uint32_t index_mask = self->capacity - 1;
509
510 uint32_t index = key_hash & index_mask;
511 uint32_t max_possible_offset = 0;
512
513 while (true) {
514 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
515
516 const bool below_max = max_possible_offset <= self->slots[index].offset;
517
518 if (!(not_empty && below_max)) {
519 break;
520 }
521
522 if (KEY_IS_EQUAL(self->slots[index].key, key)) {
523 return true;
524 }
525
526 index++;
527 index &= index_mask;
528 max_possible_offset++;
529 }
530 return false;
531}

608FUNCTION_LINKAGE void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE *self, KEY_TYPE key, VALUE_TYPE value)
609{
610 assert(self != NULL);
611 assert(FHASHTABLE_CONTAINS_KEY(self, key) == false);
612
613 const uint32_t index_mask = self->capacity - 1;
614 const uint32_t key_hash = HASH_FUNCTION(key);
615
616 uint32_t index = key_hash & index_mask;
617 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
618
619 while (true) {
620 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
621
622 if (!not_empty) {
623 break;
624 }
625
626 if (current_slot.offset > self->slots[index].offset) {
627 FHASHTABLE_SWAP_SLOTS(&self->slots[index], &current_slot);
628 }
629
630 index++;
631 index &= index_mask;
632 current_slot.offset++;
633 }
634 self->slots[index] = current_slot;
635 self->count++;
636}

638FUNCTION_LINKAGE void JOIN(FHASHTABLE_NAME, update)(FHASHTABLE_TYPE *self, KEY_TYPE key, VALUE_TYPE value)
639{
640 assert(self != NULL);
641
642 const uint32_t index_mask = self->capacity - 1;
643 const uint32_t key_hash = HASH_FUNCTION(key);
644
645 uint32_t index = key_hash & index_mask;
646 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
647
648 while (true) {
649 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
650
651 if (!not_empty) {
652 break;
653 }
654
655 const bool offset_is_same = current_slot.offset == self->slots[index].offset;
656
657 const bool key_is_equal = KEY_IS_EQUAL(current_slot.key, self->slots[index].key);
658
659 if (offset_is_same && key_is_equal) {
660 self->slots[index].value = current_slot.value;
661 return;
662 }
663
664 if (current_slot.offset > self->slots[index].offset) {
665 FHASHTABLE_SWAP_SLOTS(&current_slot, &self->slots[index]);
666 }
667
668 index++;
669 index &= index_mask;
670 current_slot.offset++;
671 }
672
673 self->slots[index] = current_slot;
674 self->count++;
675}

678static inline void JOIN(internal, JOIN(FHASHTABLE_NAME, backshift))(FHASHTABLE_TYPE *self, const uint32_t index_mask,
679 uint32_t index)
680{
681 assert(self);
682
683 uint32_t next_index = (index + 1) & index_mask;
684
685 while (true) {
686 const bool not_empty = self->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
687
688 const bool offset_is_non_zero = self->slots[next_index].offset > 0;
689
690 if (!(not_empty && offset_is_non_zero)) {
691 break;
692 }
693
694 self->slots[index] = self->slots[next_index];
695 self->slots[index].offset--;
696
697 self->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
698
699 index = next_index;
700 next_index = (index + 1) & index_mask;
701 }
702}

705FUNCTION_LINKAGE bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE *self, const KEY_TYPE key)
706{
707 assert(self != NULL);
708
709 const uint32_t index_mask = self->capacity - 1;
710 const uint32_t key_hash = HASH_FUNCTION(key);
711
712 uint32_t index = key_hash & index_mask;
713 uint32_t max_possible_offset = 0;
714
715 while (true) {
716 const bool not_empty = self->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET;
717
718 const bool below_max = max_possible_offset <= self->slots[index].offset;
719
720 if (!(not_empty && below_max)) {
721 break;
722 }
723
724 const bool key_is_equal = KEY_IS_EQUAL(key, self->slots[index].key);
725
726 if (!key_is_equal) {
727 index = (index + 1) & index_mask;
728 max_possible_offset++;
729 continue;
730 }
731
732 self->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
733 self->count--;
734
735 FHASHTABLE_BACKSHIFT(self, index_mask, index);
736
737 return true;
738 }
739 return false;
740}
