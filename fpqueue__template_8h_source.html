471static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE *self, uint32_t index)
472{
473 assert(self != NULL);
474 assert(index < self->count);
475
476 uint32_t parent;
477 while (index > 0) {
478 parent = FPQUEUE_PARENT(index);
479
480 const bool sorted = self->elements[parent].priority >= self->elements[index].priority;
481
482 if (sorted) {
483 break;
484 }
485
486 const FPQUEUE_ELEMENT_TYPE temp = self->elements[index];
487 self->elements[index] = self->elements[parent];
488 self->elements[parent] = temp;
489
490 index = parent;
491 }
492}
493
494static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE *self, const uint32_t index)
495{
496 assert(self != NULL);
497 assert(self->count == 0 || index < self->count);
498
499 const uint32_t l = FPQUEUE_LEFT_CHILD(index);
500 const uint32_t r = FPQUEUE_RIGHT_CHILD(index);
501
502 uint32_t largest = index;
503 if (l < self->count && self->elements[l].priority > self->elements[index].priority) {
504 largest = l;
505 }
506 if (r < self->count && self->elements[r].priority > self->elements[largest].priority) {
507 largest = r;
508 }
509
510 if (largest == index) {
511 return;
512 }
513
514 const FPQUEUE_ELEMENT_TYPE temp = self->elements[index];
515 self->elements[index] = self->elements[largest];
516 self->elements[largest] = temp;
517
518 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(self, largest);
519}
