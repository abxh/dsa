331FUNCTION_LINKAGE FQUEUE_TYPE *JOIN(FQUEUE_NAME, init)(FQUEUE_TYPE *self, const uint32_t pow2_capacity)
332{
333 assert(self);
334 assert(IS_POW2(pow2_capacity));
335
336 self->begin_index = self->end_index = 0;
337 self->count = 0;
338 self->capacity = pow2_capacity;
339
340 return self;
341}

343FUNCTION_LINKAGE FQUEUE_TYPE *JOIN(FQUEUE_NAME, create)(const uint32_t min_capacity)
344{
345 if (min_capacity == 0 || min_capacity > UINT32_MAX / 2 + 1) {
346 return NULL;
347 }
348
349 const uint32_t capacity = round_up_pow2_32(min_capacity);
350
351 if (FQUEUE_CALC_SIZEOF_OVERFLOWS(FQUEUE_NAME, capacity)) {
352 return NULL;
353 }
354
355 const uint32_t size = FQUEUE_CALC_SIZEOF(FQUEUE_NAME, capacity);
356
357 FQUEUE_TYPE *self = (FQUEUE_TYPE *)calloc(1, size);
358
359 if (!self) {
360 return NULL;
361 }
362
363 FQUEUE_INIT(self, capacity);
364
365 return self;
366}

422FUNCTION_LINKAGE bool JOIN(FQUEUE_NAME, enqueue)(FQUEUE_TYPE *self, const VALUE_TYPE value)
423{
424 assert(self != NULL);
425 assert(!FQUEUE_IS_FULL(self));
426
427 const uint32_t index_mask = (self->capacity - 1);
428
429 self->values[self->end_index] = value;
430 self->end_index++;
431 self->end_index &= index_mask;
432 self->count++;
433
434 return true;
435}

437FUNCTION_LINKAGE VALUE_TYPE JOIN(FQUEUE_NAME, dequeue)(FQUEUE_TYPE *self)
438{
439 assert(self != NULL);
440 assert(!FQUEUE_IS_EMPTY(self));
441
442 const uint32_t index_mask = (self->capacity - 1);
443
444 const VALUE_TYPE value = self->values[self->begin_index];
445 self->begin_index++;
446 self->begin_index &= index_mask;
447 self->count--;
448
449 return value;
450}

460FUNCTION_LINKAGE void JOIN(FQUEUE_NAME, copy)(FQUEUE_TYPE *restrict dest_ptr, const FQUEUE_TYPE *restrict src_ptr)
461{
462 assert(src_ptr != NULL);
463 assert(dest_ptr != NULL);
464 assert(src_ptr->count <= dest_ptr->capacity);
465 assert(FQUEUE_IS_EMPTY(dest_ptr));
466
467 const uint32_t src_begin_index = src_ptr->begin_index;
468 const uint32_t src_index_mask = src_ptr->capacity - 1;
469
470 for (uint32_t i = 0; i < src_ptr->count; i++) {
471 dest_ptr->values[i] = src_ptr->values[(src_begin_index + i) & src_index_mask];
472 }
473
474 dest_ptr->count = src_ptr->count;
475 dest_ptr->begin_index = 0;
476 dest_ptr->end_index = src_ptr->count;
477}
