312FUNCTION_LINKAGE FSTACK_TYPE *JOIN(FSTACK_NAME, init)(FSTACK_TYPE *self, const uint32_t capacity)
313{
314 assert(self);
315
316 self->count = 0;
317 self->capacity = capacity;
318
319 return self;
320}

322FUNCTION_LINKAGE FSTACK_TYPE *JOIN(FSTACK_NAME, create)(const uint32_t capacity)
323{
324 if (capacity == 0 || FSTACK_CALC_SIZEOF_OVERFLOWS(FSTACK_NAME, capacity)) {
325 return NULL;
326 }
327
328 const uint32_t size = FSTACK_CALC_SIZEOF(FSTACK_NAME, capacity);
329
330 FSTACK_TYPE *self = (FSTACK_TYPE *)calloc(1, size);
331
332 if (!self) {
333 return NULL;
334 }
335
336 FSTACK_INIT(self, capacity);
337
338 return self;
339}

391FUNCTION_LINKAGE void JOIN(FSTACK_NAME, push)(FSTACK_TYPE *self, const VALUE_TYPE value)
392{
393 assert(self != NULL);
394 assert(FSTACK_IS_FULL(self) == false);
395
396 self->values[self->count++] = value;

399FUNCTION_LINKAGE VALUE_TYPE JOIN(FSTACK_NAME, pop)(FSTACK_TYPE *self)
400{
401 assert(self != NULL);
402 assert(FSTACK_IS_EMPTY(self) == false);
403
404 return self->values[--self->count];
405}

407FUNCTION_LINKAGE void JOIN(FSTACK_NAME, clear)(FSTACK_TYPE *self)

409 assert(self != NULL);
410 self->count = 0;
411}

413FUNCTION_LINKAGE void JOIN(FSTACK_NAME, copy)(FSTACK_TYPE *restrict dest_ptr, const FSTACK_TYPE *restrict src_ptr)
414{
415 assert(src_ptr != NULL);
416 assert(dest_ptr != NULL);
417 assert(src_ptr->count <= dest_ptr->capacity);
418 assert(FSTACK_IS_EMPTY(dest_ptr));
419
420 for (uint32_t i = 0; i < src_ptr->count; i++) {
421 dest_ptr->values[i] = src_ptr->values[i];
422 }
423 dest_ptr->count = src_ptr->count;
424}
