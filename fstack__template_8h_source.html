321FUNCTION_LINKAGE FSTACK_TYPE *JOIN(FSTACK_NAME, init)(FSTACK_TYPE *self, const uint32_t capacity)
322{
323 assert(self);
324
325 self->count = 0;
326 self->capacity = capacity;
327
328 return self;
329}

331FUNCTION_LINKAGE FSTACK_TYPE *JOIN(FSTACK_NAME, create)(const uint32_t capacity)
332{
333 if (capacity == 0 || FSTACK_CALC_SIZEOF_OVERFLOWS(FSTACK_NAME, capacity)) {
334 return NULL;
335 }
336
337 const uint32_t size = FSTACK_CALC_SIZEOF(FSTACK_NAME, capacity);
338
339 FSTACK_TYPE *self = (FSTACK_TYPE *)calloc(1, size);
340
341 if (!self) {
342 return NULL;
343 }
344
345 FSTACK_INIT(self, capacity);
346
347 return self;
348}

400FUNCTION_LINKAGE void JOIN(FSTACK_NAME, push)(FSTACK_TYPE *self, const VALUE_TYPE value)
401{

422FUNCTION_LINKAGE void JOIN(FSTACK_NAME, copy)(FSTACK_TYPE *restrict dest_ptr, const FSTACK_TYPE *restrict src_ptr)
423{
424 assert(src_ptr != NULL);
425 assert(dest_ptr != NULL);
426 assert(src_ptr->count <= dest_ptr->capacity);
427 assert(FSTACK_IS_EMPTY(dest_ptr));
428
429 for (uint32_t i = 0; i < src_ptr->count; i++) {
430 dest_ptr->values[i] = src_ptr->values[i];
431 }
432 dest_ptr->count = src_ptr->count;
433}
